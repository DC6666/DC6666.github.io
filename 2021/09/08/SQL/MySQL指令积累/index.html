MySQL对数据的简单操作

登录于创建

登录用户

mysql -u username -p;   #;是一条命令结束的标志。

创建一个数据库

create database 数据库名;

删除某个数据库

drop database 数据库名;

显示数据库

show databases;

使用某个数据库

use database_name;

显示该数据库中的表

show tables;

在使用某个数据库后在其中新建表

CREATE TABLE IF NOT EXISTS table_name(  #这里没有';'是用的回车换行
   字段 INT UNSIGNED AUTO_INCREMENT,   #auto_inrement在插入数据时会自动插入，自动添加。
   字段 VARCHAR(100) NOT NULL,
   字段 VARCHAR(40) NOT NULL,
   字段 DATE,
   PRIMARY KEY ( 字段 )//这里设置某个字段对于不同条的数据不能相同
)ENGINE=InnoDB DEFAULT CHARSET=utf8;   #分号结束这条命令

显示表中数据

select * from table_name;

使用某个数据库后删除其中的表

drop table table_name;

插入数据

向表中插入数据(tips:插入数据的字段需要于表中的字段对应 )

insert into table_name
(字段1,字段2,字段3...)    #...是省略的意思，后面有多少数据就写多少。
values
(字段1的值,字段2的值,字段3的值...);   #该行可以不加';'而书写多行末尾加','，最后一行添加';',用于添加多组数据。

也可以直接在没有设置自动添加时直接添加对应字段的值

insert into table_name
values
(字段1的值,字段2的值,字段3的值...)

如果第一个为自动添加的数据

insert into 'table_name'
(字段2,字段3...)    #...是省略的意思，后面有多少数据就写多少。
values
(字段2的值,字段3的值...);   #字段一的值系统自动添加，比如id，前面有一个id=99，则自动添加该行id=100;

查找数据

查寻两个表的数据，并将相关属性相等的保留

select * from table1,bable2   #这句会执行乘积运算执行
where table2.属性=table2.属性;   #筛选乘积运算的数据

稍微完整的语法

SELECT column_name,column_name
FROM table_name
[WHERE Clause]   #条件
[LIMIT N][ OFFSET M];   #limit限制查找条数，offset设置偏移量，默认为0

例如：

执行

mysql> select * from test;

结果为：


+----+------+------+------------+
| id | name | 性别 | date       |
+----+------+------+------------+
|  1 | a    | 女   | 2021-09-10 |
|  2 | bfsd | 男   | 2021-09-10 |
|  3 | c    | 男   | 2021-09-10 |
|  4 | d    | 女   | 2021-09-10 |
|  5 | qq   | 男   | 2021-09-10 |
|  6 | qq   | 男   | 2021-09-10 |
+----+------+------+------------+

执行

mysql> select * from test limit 3 OFFSET 2;

结果为：


+----+------+------+------------+
| id | name | 性别 | date       |
+----+------+------+------------+
|  3 | c    | 男   | 2021-09-10 |
|  4 | d    | 女   | 2021-09-10 |
|  5 | qq   | 男   | 2021-09-10 |
+----+------+------+------------+

从第三条开始查找3条记录。值得注意的时offset要和limit一起使用。

update更新

UPDATE table_name SET field1=newvalue
[WHERE Clause];

delete删除

DELETE FROM table_name [WHERE Clause]

like子句

like可以用于where子句，使用like子句替代’=’，可以再like子句中使用’%’，类似于一个元字符的搜索。

还可以使用’and’和’or’指定一个或多个条件。

再’delete’和’update’中同样可以使用where…like子句。

简单的语法如下：

select * from 'table_name' where name like '条件';

如：

#先来看看表中的数据。
mysql> select * from test;
+----+----------+------+------------+
| id | name     | 性别 | date       |
+----+----------+------+------------+
|  1 | fds.com  | 女   | 2021-09-10 |
|  2 | gerf.com | 男   | 2021-09-10 |
|  3 | c        | 男   | 2021-09-10 |
|  4 | 哈哈哈   | 女   | 2021-09-10 |
|  5 | qq       | 男   | 2021-09-10 |
+----+----------+------+------------+
#选择name中带有'.com'的数据。
mysql> select * from test where name like '%.com';
+----+----------+------+------------+
| id | name     | 性别 | date       |
+----+----------+------+------------+
|  1 | fds.com  | 女   | 2021-09-10 |
|  2 | gerf.com | 男   | 2021-09-10 |
+----+----------+------+------------+
#选择name中带有'.com'和'q'的数据。
mysql> select * from test where name like '%.com' or name like '%q';
+----+----------+------+------------+
| id | name     | 性别 | date       |
+----+----------+------+------------+
|  1 | fds.com  | 女   | 2021-09-10 |
|  2 | gerf.com | 男   | 2021-09-10 |
|  5 | qq       | 男   | 2021-09-10 |
+----+----------+------+------------+

like 匹配/模糊匹配，会与 % 和 _ 结合使用。

'%a'     #以a结尾的数据
'a%'     #以a开头的数据
'%a%'    #含有a的数据
'_a_'    #三位且中间字母是a的
'_a'     #两位且结尾字母是a的
'a_'     #两位且开头字母是a的

MySQL对用户的操作

查询用户

#再系统默认的MySQL数据库中的user表中存有用户信息。
#显示所有用户
select user from mysql.user;
#显示本地登录用户
select user,host from mysql.user;

创建用户

#仅能本地登录用户
create user '用户名'@'localhost' identified by '密码';
#远程登录用户
create user '用户名'@'%' identified by '密码';#可以将%替换为其他ip试试。

如

create user 'test'@'localhost' identified by '123456';
