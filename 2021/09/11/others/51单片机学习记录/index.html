简单认识普中51单片机的8个灯

在A2开发版中，P2管脚的结构如图

点亮一个灯

其中1234位高电位（1），当5678位低电位（0）时，对应的灯就会亮

#include "reg52.h"
sbit led1 = P2^0;  //将P2.0管脚定义为led1
void main(){
    led1 = 0;   //当led1为低电位(0)时，第一颗灯常亮。
    while(1);
}

在开发时，为了方便，使用如下定义

typedef unsigned int u16;   //int 为两个字节
typedef unsigned char u8;   //char 为一个字节

延时函数

typedef unsigned int u16;
typedef unsigned char u8;
void delay(u16 times){   //使用时time不应该大于65535.
    while(times--)
}

灯珠闪烁

#include "reg52.h"
sbit led1 = P2^0;
void main(){
    while(1){
        led1 = 0;
        delay(50000);
        led1 = 0;
        delay(50000);
    }
}

LED流水灯

#include "reg52.h"
#define led P2
typedef unsigned int u16;
typedef unsigned char u8;
void delay(u16 time)
{
	while(time)time--;
}
void main()
{
	led = 0xfe;
	while(1)
	{
		led = (led<<1)+1;
		delay(50000);
		if(led==0x7f)
		{
			led = 0xfe;
			delay(50000);
		}
	}	
}

2.改进版

#include "reg52.h"
#define led P2
typedef unsigned int u16;
typedef unsigned char u8;
void delay(u16 time)
{
	while(time)time--;
}
void main()
{
	while(1){
		u8 i=0;
		for(i;i<8;i++)
		{
			led = ~(0x1<<i);//此处将1左移i,使得8位中只有一位等于1，再取反就得到一个为0;
			delay(50000);
		}
	}
}

3使用左移_crol_ 右移、_cror_ 函数

#include "reg52.h"
#include "intrins.h"
#define led P2
typedef unsigned int u16;
typedef unsigned char u8;
void delay(u16 time)
{
	while(time)time--;
}
void de(){
	delay(50000);
}
void main()
{
	u8 i = 0;
	led = ~0x01;
	delay(50000);
	while(1)
	{
		for(i=0;i<7;i++)
		{
			led = _crol_(led,1);//每次左移1，也可以设置2等等
			de();
		}
		for(i=0;i<7;i++)
		{
			led = _cror_(led,1);//右移1
			de();
		}
	}	
}
