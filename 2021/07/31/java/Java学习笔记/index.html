接口

所有方法都为抽象方法

关键词：interface

格式：

/*修饰符*/interface Test1 /*extends 接口1，接口2...*/{
	/*修饰符*/ /*static*//*final*/ 常量类型 常量名 = 常量;
	/*public*/ /*abstract*/ 返回值 方法名();//默认自动添加public static final
	//默认方法
	/*public*/ default 返回值 方法名() {//默认方法和静态方法自动添加public
		//默认方法体;
	}
	/*publid*/ static 返回值 方法名() {
		//静态方法方法体
	}
}

interface Test1 {
	void fun1();

	static void fun2() {
		System.out.println("静态方法");
	}

	default void fun3() {
		System.out.println("默认方法");
	}
}

interface Test2 extends Test1 {
	void fun4();
}

class Test3 implements Test2 {
	// 接口是public省略，此处不可省略
	public void fun1() {
		System.out.println("实现父类1的fun方法");
	}

	public void fun4() {
		System.out.println("实现父类2的fun4方法");
	}
}

public class Learn {
	public static void main(String[] args) {
		Test3 test = new Test3();
		test.fun1();
		Test1.fun2();// 此处静态方法直接访问;
		test.fun3();
		test.fun4();
	}
}
