package test.cn;


import java.io.IOException;
import java.io.OutputStream;
import java.net.InetAddress;
import java.net.Socket;

public class TcpDemo1 {
    public static void main(String[] args) {
        OutputStream os=null;
        Socket scoket=null;
        try 
            InetAddress serverIp = InetAddress.getByName("127.0.0.1");
            int port = 9999;
            //创建流套接字并将其连接到指定IP地址处的指定端口号。
           scoket = new Socket(serverIp,port);
            //将流交给os输出流管理
            os = scoket.getOutputStream();
            //向流输出字节数据
            os.write("这就离谱".getBytes());
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            if(os!=null){
                try {
                    os.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if(scoket!=null){
                try {
                    scoket.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}

package test.cn;

import java.io.ByteArrayOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.net.ServerSocket;
import java.net.Socket;

public class TcpServer {
    public static void main(String[] args) {
        ServerSocket serverSocket=null;
        Socket socket=null;
        InputStream is=null;
        ByteArrayOutputStream baos=null;
        try{
            serverSocket = new ServerSocket(9999);
            while(true) {
                //侦听对此套接字的连接并接受它。 该方法将阻塞，直到建立连接。创建一个新的Socket s
                socket = serverSocket.accept();
                //流托管
                is = socket.getInputStream();
                //System.out.println(is.toString());
                baos = new ByteArrayOutputStream();
                byte[] buffer = new byte[1024];
                int len = 0;
                //从流中读取数据道buffer,返回读取的长度
                while ((len = is.read(buffer)) != -1) {
                    //从buffer中读取len的数据道输出流中
                    baos.write(buffer, 0, len);
                }
                //将缓冲区的内容转换为字符串解码字节。
                String s = baos.toString();
                if (s.equals("exit"))
                    break;
                System.out.println(s);
            }
        }catch (Exception e){
            e.printStackTrace();
        }finally {
            if(baos!=null){
                try {
                    baos.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if(is!=null){
                try {
                    is.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            if(socket!=null){
                try {
                    socket.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }

    }
}

package test.cn;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetSocketAddress;

public class UdpDemo1 {
    public static void main(String[] args) throws  Exception{
        //构造一个数据报套接字并将其绑定到本地主机上的指定端口。 套接字将绑定到wildcard地址，该地址是内核选择的IP地址。
        DatagramSocket socket = new DatagramSocket(6666);
        //创建使用默认大小的输入缓冲区的缓冲字符输入流。
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));//v创建一个使用默认字符集的InputStreamReader。
        while(true) {
            //读一行文字
            String data = reader.readLine();
            byte[] datas = data.getBytes();
            //构造一个数据报包，用于将长度为 length且偏移量为offset的数据包发送到指定主机上的指定端口号
            DatagramPacket packet = new DatagramPacket(datas, 0, datas.length, new InetSocketAddress("localhost", 8888));
            //发送数据包。
            socket.send(packet);
            if(data.equals("bye"))break;
        }
        socket.close();
    }
}

package test.cn;
import java.net.DatagramPacket;
import java.net.DatagramSocket;


public class UdpServer {
    public static void main(String[] args) throws Exception {
        DatagramSocket socket = new DatagramSocket(8888);
        while(true){
            //准备接收
            byte[] container = new byte[1024];
            DatagramPacket packet = new DatagramPacket(container,0,container.length);
            socket.receive(packet);
            //断开
            byte[] data = packet.getData();//返回数据缓冲区。 收到的数据或要发送的数据从缓冲区中的offset开始，运行时间为length 。
            String receiveData = new String(data,0, packet.getLength());
            System.out.println(receiveData);
            if(receiveData.equals("bye")){
                break;
            }
        }
        socket.close();
    }
}
