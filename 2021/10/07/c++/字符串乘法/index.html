解决乘积数值越界问题

当遇到超级大的数相乘时，其结果值可能远超2100 (比如一个较大的数的阶乘)的时候，简单的数据类型如int long long long 等是无法存储的，但是可以用一个字符串来存储。

C++版本

#include <iostream>
#include <string>
#include <vector>
using namespace std;
string factorial(string num1, string num2) {
	int m = num1.length();
	int n = num1.length();
	vector <int> sum(m + n, 0);//两数相乘位数最多为两数的位数相加,开m+n的空间，初始值为0.
	for (int i{ m - 1 }; i >= 0; i--) {
		for (int j{ n - 1 }; j >= 0; j--) {
			int tmp = (num1[i] - '0') * (num2[j] - '0') + sum[j + i + 1];
			sum[j + i + 1] = tmp % 10;
			sum[j + i] += tmp / 10;
		}
	}
	//将结果转换位字符串
	string s;
	int i{ 0 };
	for (i; i < sum.size() && sum[i] == 0; i++);//在sum的最开始可能又0，不转换位字符串
	for (i; i < sum.size(); i++) {
		s.push_back(sum[i] + '0');
	}
	//如果结果位0，则s为空
	if (s.empty()) {
		return "0";
	}
	return s;
}
int main() {
	string s1, s2;
	cin >> s1 >> s2;
	string sum = factorial(s1, s2);
	cout << sum;
}

标准C版本

#include <stdio.h>
#include <string.h>
#include <malloc.h>
char* factorial ( char* p1,char* p2 ){
    int m = strlen(p1);
    int n = strlen(p2);
    int* sum = (int*)malloc(sizeof(int)*(n+m));
    for(int i = 0;i<(m+n);i++){
        sum[i]=0;
    }
    for(int i=m-1;i>=0;i--){
        for(int j = n-1;j>=0;j--){
            int tmp = (p1[i]-'0')*(p2[j]-'0')+sum[i+j+1];
            sum[i+j+1] = tmp%10;
            sum[i+j] += tmp/10;
        }
    }
    int i=0;
    while(i<m+n&&sum[i]==0)i++;
    char* re = (char*)malloc(sizeof(char)*(m+n-i+1));//不分配前面0的空间，但是多一个char的空间
    int j=0;
    while(i<n+m){
        re[j] = sum[i]+'0';
        i++;
        j++;
    }
    free(sum);
    re[j]='\0';
    return re;
}
int main(){
    char a[10];
    char b[10];
    scanf("%s %s",a,b);
    char* p = factorial(a,b);
    printf("%s",p);
    free(p);
    return 0;
}
