匿名对象和对象拷贝

#include <iostream>

class Circle {
	//public才可以访问;
public:
	double r;
	//Circle() = default;可以设置强制生成默认构造函数
	Circle() {
		r = 1.0;
	}
	Circle(double r) {
		this->r = r;//this关键词
	}
	double getAire() {
		return 3.14 * r * r;
	}
};
int main() {
	//如果直接创建对象
	/*这是有名的对象
	Circle c1{};
	Circle c2{2};
	*/
	auto c1 = Circle{};
	auto c2 = Circle{ 2 };//创建匿名类，auto类型推断，赋值给c2
	std::cout << c1.getAire() << std::endl;//3.14
	std::cout << c2.getAire() << std::endl;//12.56
	c2 = c1;//对象拷贝，c2和c1是两个不同的对象，r相同
	std::cout << c1.getAire() << std::endl;//3.14
	std::cout << c2.getAire() << std::endl;//3.14
    std::cout << Circle{ 2.2 }.getAire() << std::endl;//使用匿名对象
	decltype (c2) de{};//decltype类型推断,使用的写或不屑{}均调用无参构造方法。
	std::cout << de.getAire();
    //tips:decltype可以用于基本类型的定义
}

如何防止都文件多次引用

这里有三种常见方法

//1.大部分编译器支持
#pragma once
//2.
//2.1 VS支持,微软自己的东西，不是c++11的
__pragma(once);
//2.2  c++11的操作符在vs中不支持，因为微软有自己的
_Pragma("once");
//3.通用
//预处理指令，在没有这个宏定义的时候才编译这个都文件内容
#ifndef  MY_HEADER_FILE_H
//编译的时候首先定义这个宏
#define MY_HEADER_FILE_H
//这里放头文件内容
#endif
