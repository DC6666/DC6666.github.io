C/C++ 二维数组动态内存分配

2021-06-19- 2022-02-06
206- 1m-
Blog-Cpp

c++

#include <iostream>

using std::cout;
using std::endl;

int main() {
	int** array = new int* [5];
	//或
	//int** array;
	//array = new int* [5];
	for (int i = 0; i < 5; i++) {
		array[i] = new int[5];
		for (int j = 0; j < 5; j++) {
			array[i][j] = i * 5 + j;
		}
	}
	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 5; j++) {
			cout << array[i][j] << " ";
		}
		cout << endl;
	}
	for (int i = 0; i < 5; i++) {
		delete array[i];
	}
	delete array;
	return 0;
}

C

#include <stdio.h>
#include <malloc.h>
int main(){
	int i,j;
	int **array;
	array=(int**)malloc(5*sizeof(int*));
	for(i=0;i<5;i++){
		array[i]=(int*)malloc(5*sizeof(int));
		for(j=0;j<5;j++){
			array[i][j]=i*5+j;
		}
	}
	for(i=0;i<5;i++){
		for(j=0;j<5;j++){
			printf("%d ",array[i][j]);
		}
		printf("\n");
	}
	//释放内存
	for(i=0;i<5;i++){
		free(array[i]);
	}
	free(array);
	return 0;
}
